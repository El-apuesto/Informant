}); dropZone.addEventListener('dragleave', () => { dropZone.classList.remove('dragover'); }); dropZone.addEventListener('drop', (e) => { e.preventDefault(); dropZone.classList.remove('dragover'); const files = e.dataTransfer.files; if (files.length > 0) { handleFile(files[0]); } }); fileInput.addEventListener('change', (e) => { if (e.target.files.length > 0) { handleFile(e.target.files[0]); } }); function handleFile(file) { const validAudio = ['audio/mpeg', 'audio/wav', 'audio/x-wav', 'audio/mp3', 'audio/ogg', 'audio/webm', 'audio/aac', 'audio/m4a', 'audio/x-m4a', 'audio/flac']; const validVideo = ['video/mp4', 'video/webm', 'video/ogg', 'video/quicktime', 'video/x-msvideo', 'video/x-matroska', 'video/avi', 'video/flv', 'video/wmv']; if (!validAudio.includes(file.type) && !validVideo.includes(file.type)) { showToast('Invalid file type. Please upload audio or video files only.', 'error'); return; } currentFile = file; showFileSelected(file); } function showFileSelected(file) { const isVideo = file.type.startsWith('video/'); const fileSize = (file.size / 1024 / 1024).toFixed(2); dropContent.innerHTML = `
${file.name}
${fileSize} MB
`; actionButtons.style.display = 'flex'; errorMessage.style.display = 'none'; } // Transcribe transcribeBtn.addEventListener('click', async () => { if (!currentFile) return; transcribeBtn.disabled = true; transcribeBtn.innerHTML = '
